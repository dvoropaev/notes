1)Создать bridge, добавить в него сетевой интерфейс:

# Нам потребуется пакет bridge-utils
apt-get install bridge-utils

# Перед настройками сетевой интерфеис нужно потушить
ifconfig имя_вашего_адаптера down
# или ip link set dev eth1 down

# Создаем мост
# Если потребуется удалить bridge, то: "brctl "
brctl addbr "имя_моста"

# Необходимо выполнить ПЕРЕД дополнением интерфейса в мост
# Я пока не разобрался зачем, но если не сделать, то основная система
# не будет иметь доступа к внешней сети
ifconfig имя_интерфейса 0.0.0.0

#Добавляем основной интерфейс в мост
brctl addif имя_моста имя_интерфейса

#Получаем настройки для моста по DHCP

dhcpcd bridge
# Теперь основным сетевым интерфейсом на хосте будет мост
# если есть какие-то правила iptables и т.п, то это надо учитывать

2)Добавить в /etc/qemu/bridge.conf (Если файла нет, то зоздать)  allow имя_моста

3)подключить виртуальную машину к мосту:
# Для этого, к параметрам qemu-system добавить следующие параметры:
# имя_виртуального_адаптера произвольное
# MAC произвольный, но первая его половина 52:54:00: специально зарезервирована для qemu\kvm.


qemu-system- ... -netdev имя_моста,id=имя_виртуального_адаптера,br=имя_моста -device virtio-net-pci,netdev=имя_виртуального_адаптера,mac=52:54:00:12:34:56
